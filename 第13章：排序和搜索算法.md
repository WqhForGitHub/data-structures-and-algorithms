### 1. 排序算法

#### 1. 冒泡排序

```javascript
function bubbleSort(array) {
    const { length } = array;
    for (let i = 0; i < length; i++) {
        for (let j = 0; j < length - 1 - i; j++) {
            if(array[j] > array[j + 1]) {
                [array[j], array[j + 1]] = [array[j + 1], array[j]];
            }
        }
    }
    
    return array;
}
```
